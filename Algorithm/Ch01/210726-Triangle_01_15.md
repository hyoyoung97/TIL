### > 문제
* 4가지의 직각삼각형 모양으로 나타냄

### > 출력예시
단수는 -> 3

```
삼각형 모양으로 나타냅니다.
단수는 ：3
왼쪽 아래가 직각인 삼각형
*
**
***
왼쪽 위가 직각인 삼각형
***
**
*
오른쪽 위가 직각인 삼각형
***
 **
  *
오른쪽 아래가 직각인 삼각형
  *
 **
***
```

[for 의 i와 j의 기준을 0으로 잡을 때 - 나의 답]
```java
package chap01;
//4가지의 직각삼각형 모양으로 나타냄

import java.util.Scanner;

public class Triangle_01_15 {
	
	static void triangleLB(int n) {		//왼쪽 아래가 직각인 삼각형
		for (int i = 0; i < n; i++) {
			int a = n - (n - i) + 1;
			for (int j = 0; j < a; j++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}
	
	static void triangleLU(int n) {		//왼쪽 위가 직각인 삼각형
		for (int i = 0; i < n; i++) {
			int a = n - i;
			for (int j = 0; j < a; j++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}
	
	static void triangleRU(int n) {		//오른쪽 위가 직각인 삼각형
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < i ; j++) {
				System.out.print(" ");
			}
			int b = n - i;
			for (int j = 0; j < b; j++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}
	
	static void triangleRB(int n) {		//오른쪽 아래가 직각인 삼각형
		for (int i = 0; i < n; i++) {
			int a = n - 1 - i;
			for (int j = 0; j < a ; j++) {
				System.out.print(" ");
			}
			int b = n - (n - i) + 1;
			for (int j = 0; j < b; j++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}
	

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		
		System.out.println("삼각형 모양으로 나타냅니다.");
		
		int n;
		do {
		System.out.print("단수는 ：");
		n = sc.nextInt();
		} while(n <= 0);
		
		System.out.println("왼쪽 아래가 직각인 삼각형");
		triangleLB(n);
		
		System.out.println("왼쪽 위가 직각인 삼각형");
		triangleLU(n);
		
		System.out.println("오른쪽 위가 직각인 삼각형");
		triangleRU(n);

		System.out.println("오른쪽 아래가 직각인 삼각형");
		triangleRB(n);

	}
}
```

package chap01;

import java.util.Scanner;

public class Triangle_01_15_2 {

	static void triangleLB(int n) {		//왼쪽 아래가 직각인 삼각형
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= i; j++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}
	
	static void triangleLU(int n) {		//왼쪽 위가 직각인 삼각형
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n - i + 1 ; j++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}
	
	static void triangleRU(int n) {		//오른쪽 위가 직각인 삼각형
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <=  i - 1; j++) {
				System.out.print(" ");
			}
			for (int j = 1; j <= n - i + 1; j++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}
	
	static void triangleRB(int n) {		//오른쪽 아래가 직각인 삼각형
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n - i; j++) {
				System.out.print(" ");
			}
			for (int j = 0; j < i; j++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		
		System.out.println("삼각형 모양으로 나타냅니다.");
		
		int n;
		do {
		System.out.print("단수는 ：");
		n = sc.nextInt();
		} while(n <= 0);
		
		System.out.println("왼쪽 아래가 직각인 삼각형");
		triangleLB(n);
		
		System.out.println("왼쪽 위가 직각인 삼각형");
		triangleLU(n);
		
		System.out.println("오른쪽 위가 직각인 삼각형");
		triangleRU(n);

		System.out.println("오른쪽 아래가 직각인 삼각형");
		triangleRB(n);
	}

}
