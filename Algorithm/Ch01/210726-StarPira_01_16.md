### > 문제
* 기호문자 *로 피라미드를 출력함

### > 출력예시
단수는 -> 3

```
피라미드 모양으로 나타냅니다.
단수는 ：
3
   *
  ***
 *****
```

[for 의 i와 j의 기준을 0으로 잡을 때 - 나의 답]

```java
package chap01;

import java.util.Scanner;

//기호문자 *로 피라미드를 출력함

public class StarPira_01_16_2 {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int n;
		System.out.println("피라미드 모양으로 나타냅니다.");
		
		do {
		System.out.println("단수는 ：");
		n = sc.nextInt();
		} while (n <= 0);
		
		for(int i = 0; i < n; i++) {
			for(int j = 0; j < n - i; j++) {
				System.out.print(" ");
			}
			for(int j = 0; j < i + (i + 1); j++) {
				System.out.print("*");
			}
			System.out.println();
		}
		
	}

}
```

package chap01;

import java.util.Scanner;

//기호문자 *로 피라미드를 출력함

public class StarPira_01_16 {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int n;
		System.out.println("피라미드 모양으로 나타냅니다.");
		
		do {
		System.out.println("단수는 ：");
		n = sc.nextInt();
		} while (n <= 0);
		
		for(int i = 1; i <= n; i++) {
			for(int j = 1; j <= n - i + 1; j++) {
				System.out.print(" ");
			}
			for(int j = 1; j <= i + (i - 1); j++) {
				System.out.print("*");
			}
			System.out.println();
		}
		
	}

}

[for 의 i와 j의 기준을 1으로 잡을 때 - 답안지]

```java
package chap01;
import java.util.Scanner;
// 기호문자 *로 피라미드를 출력함

class StarPira_01_16 {
	static void spira(int n) {
		for (int i = 1; i <= n; i++) { 					// i행 (i = 1, 2, … ,n)
			for (int j = 1; j <= n - i + 1; j++) 		// n-i+1개의 ' '를 나타냄
				System.out.print(' ');
			for (int j = 1; j <= (i - 1) * 2 + 1; j++) 	// (i-1)*2+1개의 '*'를 나타냄
				System.out.print('*');
			System.out.println(); 						// 개행(줄변환)
		}
	}

	public static void main(String[] args) {
		Scanner stdIn = new Scanner(System.in);
		int n;

		System.out.println("피라미드 모양으로 나타냅니다.");

		do {
			System.out.print("단수는 ：");
			n = stdIn.nextInt();
		} while (n <= 0);

		spira(n); 		// 피라미드를 나타냄
	}
}
```
