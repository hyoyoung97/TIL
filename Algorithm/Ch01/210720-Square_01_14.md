### > 문제
* 정사각형 모양을 나타냄

### > 출력예시
단수는 -> 3
```
정사각형 모양으로 나타냅니다.
단수는 : 3
***
***
***
```
[나의 답]
```java
package chap01;
//정사각형 모양을 나타냄

import java.util.Scanner;

public class Square_01_14 {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		System.out.println("정사각형 모양으로 나타냅니다.");
    
    //do-while문으로 바꾸는 것이 좋았을것
		System.out.print("단수는 : ");
		int n = sc.nextInt();	
		
		//행을 출력하는 코드 
		for(int i = 0; i < n; i++) {
			//열을 출력하는 코드
			for(int j = 0; j < n; j++) {
				System.out.print("*");
				//줄바꾸기 
			} System.out.println();
		}
	}

}
```

[답안지]
```java
package chap01;
import java.util.Scanner;
// 정사각형 모양을 나타냄

public class Square_01_14 {
	public static void main(String[] args) {
		Scanner stdIn = new Scanner(System.in);
		int n;

		System.out.println("정사각형 모양으로 나타냅니다.");

		do {
			System.out.print("단수는：");
			n = stdIn.nextInt();
		} while (n <= 0);

		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n; j++)
				System.out.print('*');
			System.out.println();
		}
	}
}
```
